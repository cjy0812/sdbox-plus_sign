// 获取指定进程的所有线程ID
std::vector&lt;DWORD&gt; GetProcessThreads(DWORD processId) {
    std::vector&lt;DWORD&gt; threadIds;

    THREADENTRY32 te;
    te.dwSize = sizeof(THREADENTRY32);

    HANDLE hSnapshot = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0);
    if (hSnapshot == INVALID_HANDLE_VALUE) {
        std::wcout &lt;&lt; L"创建快照失败!" &lt;&lt; std::endl;
        return threadIds;
    }

    // 枚举线程
    if (Thread32First(hSnapshot, &amp;te)) {
        do {
            if (te.th32OwnerProcessID == processId) {
                threadIds.push_back(te.th32ThreadID);
            }
        } while (Thread32Next(hSnapshot, &amp;te));
    }
    CloseHandle(hSnapshot);
    return threadIds;
}

// 挂起进程的所有线程
bool SuspendProcess(DWORD processId) {
    std::vector&lt;DWORD&gt; threadIds = GetProcessThreads(processId);
    for (DWORD threadId : threadIds) {
        HANDLE hThread = OpenThread(THREAD_SUSPEND_RESUME, FALSE, threadId);
        if (hThread != NULL) {
            DWORD suspendCount = SuspendThread(hThread);
            if (suspendCount == (DWORD)-1) {
                std::wcout &lt;&lt; L"[i] 进程挂起，线程ID: " &lt;&lt; threadId &lt;&lt; std::endl;
            }
            CloseHandle(hThread);
        }
        else {
            std::wcout &lt;&lt; L"[i] 无法打开线程，线程ID: " &lt;&lt; threadId &lt;&lt; std::endl;
        }
    }
    return true;
}

// 恢复进程的所有线程
bool ResumeProcess(DWORD processId) {
    std::vector&lt;DWORD&gt; threadIds = GetProcessThreads(processId);
    for (DWORD threadId : threadIds) {
        HANDLE hThread = OpenThread(THREAD_SUSPEND_RESUME, FALSE, threadId);
        if (hThread != NULL) {
            DWORD resumeCount = ResumeThread(hThread);
            if (resumeCount == (DWORD)-1) {
                std::wcout &lt;&lt; L"[i] 恢复进程挂起状态，线程ID: " &lt;&lt; threadId &lt;&lt; std::endl;
            }
            CloseHandle(hThread);
        }
        else {
            std::wcout &lt;&lt; L"[x] 恢复进程挂起失败: " &lt;&lt; threadId &lt;&lt; std::endl;
        }
    }
    return true;
}

// 获取指定进程的句柄
HANDLE GetProcessHandle(const std::wstring&amp; processName) {
    HANDLE hSnapshot = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    if (hSnapshot == INVALID_HANDLE_VALUE) return NULL;

    PROCESSENTRY32 pe;
    pe.dwSize = sizeof(PROCESSENTRY32);
    if (Process32First(hSnapshot, &amp;pe)) {
        do {
            if (!processName.compare(pe.szExeFile)) {
                CloseHandle(hSnapshot);
                return OpenProcess(PROCESS_ALL_ACCESS, FALSE, pe.th32ProcessID);

            }
        } while (Process32Next(hSnapshot, &amp;pe));
    }
    CloseHandle(hSnapshot);
    return NULL;
}

// 在进程内存中搜索字节模式
std::vector&lt;uintptr_t&gt; SearchMemory(HANDLE hProcess, const std::vector&lt;uint8_t&gt;&amp; pattern) {
    std::vector&lt;uintptr_t&gt; results;
    MEMORY_BASIC_INFORMATION mbi;
    uintptr_t address = 0;

    while (VirtualQueryEx(hProcess, (LPCVOID)address, &amp;mbi, sizeof(mbi))) {
        if (mbi.State == MEM_COMMIT &amp;&amp; (mbi.Protect &amp; (PAGE_READWRITE | PAGE_READONLY | PAGE_EXECUTE_READ))) {

            std::vector&lt;uint8_t&gt; buffer(mbi.RegionSize);
            SIZE_T bytesRead;
            if (ReadProcessMemory(hProcess, (LPCVOID)address, buffer.data(), mbi.RegionSize, &amp;bytesRead)) {
                for (size_t i = 0; i &lt; bytesRead - pattern.size(); ++i) {

                    if (memcmp(buffer.data() + i, pattern.data(), pattern.size()) == 0) {
                        results.push_back(address + i);
                    }
                }
            }
        }
        address += mbi.RegionSize;
    }
    return results;
}

bool WriteMemory(HANDLE hProcess, uintptr_t address, const std::vector&lt;uint8_t&gt;&amp; newData) {
    DWORD oldProtect;
    // 改变内存区域的保护为可写
    if (!VirtualProtectEx(hProcess, (LPVOID)address, newData.size(), PAGE_READWRITE, &amp;oldProtect)) {
        return false;
    }

    SIZE_T bytesWritten;
    if (WriteProcessMemory(hProcess, (LPVOID)address, newData.data(), newData.size(), &amp;bytesWritten)) {
        // 恢复原来的内存保护
        VirtualProtectEx(hProcess, (LPVOID)address, newData.size(), oldProtect, &amp;oldProtect);
        return bytesWritten == newData.size();
    }

    // 恢复原来的内存保护
    VirtualProtectEx(hProcess, (LPVOID)address, newData.size(), oldProtect, &amp;oldProtect);
    return false;
}

bool EnablePrivilege(LPCWSTR priv) {
    HANDLE hToken;
    TOKEN_PRIVILEGES tp;
    LUID luid;

    if (!OpenProcessToken(GetCurrentProcess(), TOKEN_ADJUST_PRIVILEGES | TOKEN_QUERY, &amp;hToken)) {
        return false;
    }

    if (!LookupPrivilegeValueW(NULL, priv, &amp;luid)) {
        CloseHandle(hToken);
        return false;
    }

    tp.PrivilegeCount = 1;
    tp.Privileges[0].Luid = luid;
    tp.Privileges[0].Attributes = SE_PRIVILEGE_ENABLED;

    if (!AdjustTokenPrivileges(hToken, FALSE, &amp;tp, 0, NULL, NULL)) {
        CloseHandle(hToken);
        return false;
    }

    CloseHandle(hToken);
    return true;
}



bool TerminateProcessByName(const std::wstring&amp; processName) {
    // 获取进程快照
    HANDLE hSnapshot = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    if (hSnapshot == INVALID_HANDLE_VALUE) {
        std::wcout &lt;&lt; L"[x] 无法获取进程快照" &lt;&lt; std::endl;
        return false;
    }

    PROCESSENTRY32 pe32;
    pe32.dwSize = sizeof(PROCESSENTRY32);

    // 遍历进程列表
    if (Process32First(hSnapshot, &amp;pe32)) {
        do {
            // 如果进程名称匹配
            if (processName == pe32.szExeFile) {
                // 打开进程
                HANDLE hProcess = OpenProcess(PROCESS_TERMINATE, FALSE, pe32.th32ProcessID);
                if (hProcess != NULL) {
                    // 终止进程
                    BOOL result = TerminateProcess(hProcess, 0);
                    CloseHandle(hProcess);
                    if (result) {
                        std::wcout &lt;&lt; L"[v] " &lt;&lt; processName &lt;&lt; L"已经被结束" &lt;&lt; std::endl;
                        CloseHandle(hSnapshot);
                        return true;
                    }
                    else {
                        std::wcout &lt;&lt; L"[x] 无法结束进程" &lt;&lt; std::endl;
                        CloseHandle(hSnapshot);
                        return false;
                    }
                }
            }
        } while (Process32Next(hSnapshot, &amp;pe32));
    }

    std::wcout &lt;&lt; L"[x] 未找到进程" &lt;&lt; std::endl;
    CloseHandle(hSnapshot);
    return false;
}
bool StartProcessWithArgs(const std::wstring&amp; exePath, const std::wstring&amp; args) {
    // 创建 STARTUPINFO 结构体，初始化为零
    STARTUPINFO si = { 0 };
    si.cb = sizeof(STARTUPINFO);

    // 创建 PROCESS_INFORMATION 结构体，初始化为零
    PROCESS_INFORMATION pi = { 0 };

    // 构造命令行
    std::wstring commandLine = exePath + L" " + args;

    // 使用 CreateProcess 启动进程
    if (CreateProcess(
        nullptr,                 // 应用程序名称，如果命令行中包含程序路径，此参数可以为 nullptr
        &amp;commandLine[0],         // 命令行字符串
        nullptr,                 // 进程安全属性
        nullptr,                 // 线程安全属性
        FALSE,                   // 是否继承句柄
        0,                       // 创建标志
        nullptr,                 // 环境变量
        nullptr,                 // 当前目录
        &amp;si,                     // STARTUPINFO
        &amp;pi                      // PROCESS_INFORMATION
    )) {
        std::wcout &lt;&lt; L"[i] 进程创建成功" &lt;&lt; std::endl;
        // 等待进程结束
        // WaitForSingleObject(pi.hProcess, INFINITE);

        // 关闭进程和线程句柄
        CloseHandle(pi.hProcess);
        CloseHandle(pi.hThread);

        return true;
    }
    else {
        std::wcout &lt;&lt; L"[x] 无法创建进程 错误代码: " &lt;&lt; GetLastError() &lt;&lt; std::endl;
        return false;
    }
}
